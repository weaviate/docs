<TabItem value="go" label="Go">

```go
package main

import (
  "context"
  "fmt"

  "github.com/weaviate/weaviate-go-client/v5/weaviate"
)

func main() {
  cfg := weaviate.Config{
    Host:   "localhost:8080",
    Scheme: "http",
  }
  client, err := weaviate.NewClient(cfg)
  if err != nil {
    panic(err)
  }

  concept, err := client.C11y().
    ConceptsGetter().
    WithConcept("fashionMagazine").
    Do(context.Background())

  if err != nil {
    panic(err)
  }

  fmt.Printf("%v", concept)
}
```
